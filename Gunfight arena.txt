-- Safe Parent Check (verhindert Fehler bei verschiedenen Executors)
local function getSafeGuiRoot()
    if gethui then return gethui() end
    if game:GetService("CoreGui") then return game:GetService("CoreGui") end
    return LocalPlayer:WaitForChild("PlayerGui")
end

--------------------------------------------------------------------------------
-- GAME LOGIC & SETTINGS
--------------------------------------------------------------------------------
local Settings = {
    Aimbot = false,
    AimPart = "Head",
    TeamCheck = true,
    ESP = false,
    NoRecoil = false
}

-- [AIMBOT LOGIK] --
local function GetTarget()
    local target = nil
    local dist = math.huge
    local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild("Humanoid") and v.Character.Humanoid.Health > 0 then
            -- Team Check
            if Settings.TeamCheck and v.Team == LocalPlayer.Team then continue end
            
            local part = v.Character:FindFirstChild(Settings.AimPart)
            if part then
                local pos, vis = Camera:WorldToViewportPoint(part.Position)
                if vis then
                    local mag = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                    if mag < dist and mag < 200 then -- 200 Pixel FOV
                        dist = mag
                        target = part
                    end
                end
            end
        end
    end
    return target
end

-- [ESP LOGIK (Optimiert für Mobile Highlight)] --
local function UpdateESP()
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= LocalPlayer and v.Character then
            local hasHighlight = v.Character:FindFirstChild("PandaESP")
            
            if Settings.ESP then
                if not hasHighlight then
                    local hl = Instance.new("Highlight")
                    hl.Name = "PandaESP"
                    hl.Parent = v.Character
                    hl.FillColor = Color3.fromRGB(255, 50, 50)
                    hl.OutlineColor = Color3.fromRGB(255, 255, 255)
                    hl.FillTransparency = 0.5
                    hl.OutlineTransparency = 0
                end
            else
                if hasHighlight then hasHighlight:Destroy() end
            end
        end
    end
end

-- Schneller Loop für Aimbot & Recoil
RunService.RenderStepped:Connect(function()
    if Settings.Aimbot then
        local target = GetTarget()
        if target then
            TweenService:Create(Camera, TweenInfo.new(0.06), {CFrame = CFrame.new(Camera.CFrame.Position, target.Position)}):Play()
        end
    end
    
    if Settings.NoRecoil then
        pcall(function()
            -- Pfad aus deiner ersten Nachricht
            if LocalPlayer.PlayerScripts.Vortex.Modifiers:FindFirstChild("Steadiness") then
                LocalPlayer.PlayerScripts.Vortex.Modifiers.Steadiness.Value = 0
            end
        end)
    end
end)

-- Langsamerer Loop für ESP Updates (Spart Akku)

    while task.wait(1) do
        if Settings.ESP then UpdateESP() end
    end
end)
