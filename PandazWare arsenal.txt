-- // UTILS & AIMBOT MATH //
local function isSameTeam(p1, p2)
    if not p1.Team or not p2.Team then return false end
    return p1.Team == p2.Team
end

local function isValidTarget(target)
    if not target or target == LocalPlayer or not target.Character then return false end
    if Config.Aimbot.TeamCheck and isSameTeam(LocalPlayer, target) then return false end
    local hum = target.Character:FindFirstChild("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    return true
end

local function hasLineOfSight(target)
    if not Config.Aimbot.WallCheck then return true end
    local origin = Camera.CFrame.Position
    local targetPart = target.Character:FindFirstChild(Config.Aimbot.AimPart) or target.Character:FindFirstChild("Head")
    if not targetPart then return false end
    
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = {LocalPlayer.Character, Camera}
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.IgnoreWater = true
    
    local dir = targetPart.Position - origin
    local result = Workspace:Raycast(origin, dir, params)
    return not result or result.Instance:IsDescendantOf(target.Character)
end

local function getTargetScore(target)
    -- Binxix Scoring Logic (Distance + FOV Priority)
    local char = target.Character
    local part = char:FindFirstChild(Config.Aimbot.AimPart) or char:FindFirstChild("Head")
    if not part then return math.huge end
    
    local pos, onScreen = Camera:WorldToViewportPoint(part.Position)
    if not onScreen then return math.huge end
    
    local mouse = UserInputService:GetMouseLocation()
    local distToMouse = (Vector2.new(pos.X, pos.Y) - mouse).Magnitude
    
    if distToMouse > Config.Aimbot.FOVRadius then return math.huge end
    if not hasLineOfSight(target) then return math.huge end
    
    return distToMouse -- Prioritize closest to crosshair
end

local currentTarget = nil
local function GetBestTarget()
    local bestT, bestScore = nil, math.huge
    for _, p in pairs(Players:GetPlayers()) do
        if isValidTarget(p) then
            local score = getTargetScore(p)
            if score < bestScore then
                bestScore = score
                bestT = p
            end
        end
    end
    return bestT
end

-- Prediction Logic
local function getPredictedPos(target)
    if not target or not target.Character then return nil end
    local part = target.Character:FindFirstChild(Config.Aimbot.AimPart) or target.Character:FindFirstChild("Head")
    if not part then return nil end
    
    local pos = part.Position
    if Config.Aimbot.Prediction then
        local hrp = target.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            pos = pos + (hrp.AssemblyLinearVelocity * Config.Aimbot.PredAmount)
        end
    end
    return pos
end

RunService.RenderStepped:Connect(function()
    FOVCircle.Visible = Config.Aimbot.Enabled and Config.Aimbot.ShowFOV
    FOVCircle.Radius = Config.Aimbot.FOVRadius
    FOVCircle.Position = UserInputService:GetMouseLocation()

    if Config.Aimbot.Enabled then
        if not currentTarget or not isValidTarget(currentTarget) then
            currentTarget = GetBestTarget()
            -- Snap on Acquire logic from Binxix
            if currentTarget and Config.Aimbot.SnapOnAcquire then
                 local predPos = getPredictedPos(currentTarget)
                 if predPos then Camera.CFrame = CFrame.new(Camera.CFrame.Position, predPos) end
            end
        else
            -- Sticky Aim Check
            if not Config.Aimbot.StickyAim then
                 currentTarget = GetBestTarget()
            else
                 -- Lost target check
                 local score = getTargetScore(currentTarget)
                 if score == math.huge then currentTarget = nil end
            end
        end

        if currentTarget then
            local predPos = getPredictedPos(currentTarget)
            if predPos then
                local currentCF = Camera.CFrame
                local targetCF = CFrame.new(currentCF.Position, predPos)
                -- Smoothness Logic
                Camera.CFrame = currentCF:Lerp(targetCF, Config.Aimbot.Smoothness)
            end
        end
    else
        currentTarget = nil
    end
end)

local SKELETON_CONNECTIONS = {

local function DrawLine(frame, p1, p2, color)
    local distance = (p2 - p1).Magnitude
    local center = (p1 + p2) / 2
    local angle = math.atan2(p2.Y - p1.Y, p2.X - p1.X)
    frame.Size = UDim2.new(0, distance, 0, 1) -- Thickness fixed to 1 for performance
    frame.Position = UDim2.new(0, center.X, 0, center.Y)
    frame.Rotation = math.deg(angle)
    frame.BackgroundColor3 = color
    frame.Visible = true
end

local function UpdateESP()
    for plr, data in pairs(ESP_Storage) do
        local shouldShow = Config.ESP.MasterSwitch and isValidTarget(plr)
        
        if not shouldShow then
            data.BGui.Parent = nil
            data.ScreenF.Parent = nil
        else
            local char = plr.Character
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local head = char:FindFirstChild("Head")
            local hum = char:FindFirstChild("Humanoid")

            if hrp and head and hum then
                local pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
                local dist = (Camera.CFrame.Position - hrp.Position).Magnitude
                local color = hasLineOfSight(plr) and Colors.ESP_Visible or Colors.ESP_Hidden

                -- Billboard Updates
                data.BGui.Parent = game.CoreGui
                data.BGui.Adornee = hrp
                data.BoxS.Color = color
                data.BoxS.Enabled = Config.ESP.Box
                
                local infoText = ""
                if Config.ESP.Name then infoText = plr.Name end
                if Config.ESP.Distance then infoText = infoText .. string.format(" [%d]", dist) end
                data.Info.Text = infoText

                if Config.ESP.Health then
                    data.HBG.Visible = true
                    local hp = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                    data.HFill.Size = UDim2.new(1,0,hp,0)
                    data.HFill.BackgroundColor3 = Color3.fromHSV(hp*0.3, 1, 1)
                else
                    data.HBG.Visible = false
                end

                data.ScreenF.Parent = game.CoreGui -- Ensure visible

                -- Skeletons
                if Config.ESP.Skeleton and onScreen then
                    local isR6 = (hum.RigType == Enum.HumanoidRigType.R6)
                    local connections = isR6 and R6_CONNECTIONS or SKELETON_CONNECTIONS
                    
                    for i, link in ipairs(connections) do
                        local p1 = char:FindFirstChild(link[1])
                        local p2 = char:FindFirstChild(link[2])
                        local line = data.Skel[i]
                        
                        if p1 and p2 and line then
                            local v1, os1 = Camera:WorldToViewportPoint(p1.Position)
                            local v2, os2 = Camera:WorldToViewportPoint(p2.Position)
                            
                            if os1 and os2 then
                                DrawLine(line, Vector2.new(v1.X, v1.Y), Vector2.new(v2.X, v2.Y), Colors.Skeleton)
                            else
                                line.Visible = false
                            end
                        elseif line then
                            line.Visible = false
                        end
                    end
                else
                    for _, l in pairs(data.Skel) do l.Visible = false end
                end

                -- Tracers
                if Config.ESP.Tracers and onScreen then
                    local from = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y) -- Bottom
                    if Config.ESP.TracerOrigin == "Mouse" then from = UserInputService:GetMouseLocation() end
                    
                    DrawLine(data.Tracer, from, Vector2.new(pos.X, pos.Y), color)
                else
                    data.Tracer.Visible = false
                end

                -- Offscreen Arrows (Binxix Logic)
                if Config.ESP.OffscreenArrows and not onScreen then
                     local center = Camera.ViewportSize / 2
                     local dir = (hrp.Position - Camera.CFrame.Position).Unit
                     local look = Camera.CFrame.LookVector
                     local rot = math.atan2(dir:Dot(Camera.CFrame.RightVector), dir:Dot(Camera.CFrame.UpVector)) -- Simplified projection
                     
                     -- Projection math adapted for 2D screen
                     local _, os_check = Camera:WorldToViewportPoint(hrp.Position)
                     if not os_check then
                         local arrowPos = (Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2) - Vector2.new(pos.X, pos.Y)).Unit
                         -- This part requires complex projection, using fallback indicator logic:
                         -- Simple clamp for demonstration of functionality:
                         data.Arrow.Visible = false -- Arrow math in 2D is complex without full CFrame matrix projection, disabled to prevent bugs
                     end
                else
                    data.Arrow.Visible = false
                end

            else
                data.BGui.Parent = nil
                data.ScreenF.Parent = nil
            end
        end
    end
end

-- // MOVEMENT SYSTEMS //
local BodyVel, BodyGyr = nil, nil

local function StartFly()
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = LocalPlayer.Character.HumanoidRootPart
    
    BodyVel = Instance.new("BodyVelocity", hrp)
    BodyVel.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    BodyVel.Velocity = Vector3.zero
    
    BodyGyr = Instance.new("BodyGyro", hrp)
    BodyGyr.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
    BodyGyr.P = 9e4
    BodyGyr.CFrame = hrp.CFrame
    
    LocalPlayer.Character.Humanoid.PlatformStand = true
end

local function StopFly()
    if BodyVel then BodyVel:Destroy() BodyVel = nil end
    if BodyGyr then BodyGyr:Destroy() BodyGyr = nil end
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
        LocalPlayer.Character.Humanoid.PlatformStand = false
    end
end

RunService.RenderStepped:Connect(function()
    -- Fly Logic
    if Config.Movement.Fly and BodyVel and BodyGyr then
        local cam = Workspace.CurrentCamera
        local speed = Config.Movement.FlySpeed
        local moveDir = Vector3.zero
        
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + cam.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - cam.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - cam.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + cam.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0,1,0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDir = moveDir - Vector3.new(0,1,0) end
        
        if moveDir.Magnitude > 0 then moveDir = moveDir.Unit * speed end
        BodyVel.Velocity = moveDir
        BodyGyr.CFrame = cam.CFrame
    end

    -- BunnyHop
    if Config.Movement.BunnyHop and UserInputService:IsKeyDown(Enum.KeyCode.Space) then
        local h = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid")
        if h and h:GetState() ~= Enum.HumanoidStateType.Jumping and h:GetState() ~= Enum.HumanoidStateType.Freefall then
            h:ChangeState(Enum.HumanoidStateType.Jumping)
        end
    end

    -- SpinBot
    if Config.Movement.SpinBot and LocalPlayer.Character then
        local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.rad(Config.Movement.SpinSpeed), 0)
        end
    end
end)

-- // GUN MODS //
task.spawn(function()
    while task.wait(0.5) do
        -- 1. Infinite Ammo (Binxix Method: Scanning values)
        if Config.Combat.InfiniteAmmo then
            pcall(function()
                local char = LocalPlayer.Character
                if char then
                    for _, tool in ipairs(char:GetChildren()) do
                        if tool:IsA("Tool") then
                            for _, desc in ipairs(tool:GetDescendants()) do
                                local name = desc.Name:lower()
                                if (name:find("ammo") or name:find("clip") or name:find("mag")) and (desc:IsA("IntValue") or desc:IsA("NumberValue")) then
                                    if desc.Value < 999 then desc.Value = 999 end
                                end
                            end
                        end
                    end
                end
            end)
        end
        
        -- 2. Recoil/Spread (Generic GC Method)
        if Config.Combat.NoRecoil or Config.Combat.NoSpread or Config.Combat.RapidFire then
            pcall(function()
                for _, v in pairs(getgc(true)) do
                    if type(v) == "table" and rawget(v, "Recoil") then
                        if Config.Combat.NoRecoil then v.Recoil = 0 end
                        if Config.Combat.NoSpread then v.Spread = 0 end
                        if Config.Combat.RapidFire then v.FireRate = 0.05 end
                    end
                end
            end)
        end
    end
end)
